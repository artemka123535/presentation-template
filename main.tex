% === СЛАЙД 2: Актуальность, проблема и цель ===
\begin{frame}
\frametitle{Актуальность и проблематика}
\begin{itemize}
    \item \textbf{Проблема:} Информационная перегрузка в стриминговых сервисах (на примере Яндекс Музыки).
    \item \textbf{Барьер:} Классические системы уязвимы к проблеме <<холодного старта>> --- новые композиции не получают охватов.
    \item \textbf{Цель:} Разработать систему на основе эмбеддингов музыкальных композиций с применением механизмов Attention для эффективного ранжирования нового контента.
\end{itemize}
\end{frame}

% === СЛАЙД 3: Анализ существующих решений ===
\begin{frame}
\frametitle{Анализ существующих решений}
\begin{itemize}
    \item \textbf{Коллаборативная фильтрация:} 
    \begin{itemize}
        \item[+] Высокая точность, учитывает сложные паттерны поведения.
        \item[-] Уязвима к <<холодному старту>> для новых треков.
    \end{itemize}
    \item \textbf{Контентные методы:}
    \begin{itemize}
        \item[+] Нет проблемы холодного старта (опирается на звук).
        \item[-] Игнорирует тренды, а также не рекомендует пользователю нечто новое для него.
    \end{itemize}
    \item \textbf{Гибридные методы (Выбранный подход):}
    \begin{itemize}
        \item[+] Объединяет акустические характеристики и социальные сигналы.
    \end{itemize}
\end{itemize}
\end{frame}

% === СЛАЙД 5: Архитектура Two-Tower ===
\begin{frame}
\frametitle{Архитектура модели: Two-Tower Neural Network}
\begin{itemize}
    \item \textbf{User Tower (Башня пользователя):} Векторизация истории профиля слушателя.
    \item \textbf{Item Tower (Башня трека):} Обработка акустических признаков трека.
    \item \textbf{Взаимодействие:} Векторы сходятся через операцию скалярного произведения (Dot Product).
\end{itemize}

% === СЛАЙД 6: Механизм Attention ===
\begin{frame}
\frametitle{Архитектура модели: Latent Sequence Attention}
\begin{itemize}
    \item \textbf{Проблема:} Простые линейные слои теряют внутреннюю структуру вектора.
    \item \textbf{Решение:} Разбиение вектора на блоки и вычисление матрицы попарных корреляций для нахождения динамических весов.
    \item \textbf{Multi-Head Attention:} Позволяет сети фокусироваться на независимых характеристиках звука (ритм, тембр).
\end{itemize}
\end{frame}

% === СЛАЙД 7: Алгоритм гибридного ранжирования ===
\begin{frame}
\frametitle{Алгоритм гибридного ранжирования}
Для учета глобальных трендов используется популярность композиций:
\begin{equation*}
Score_{hybrid} = NN_{scaled} + \alpha \cdot Pop_{scaled}
\end{equation*}
Где:
\begin{itemize}
    \item $NN_{scaled}$ --- предсказание нейросети (акустическое сходство).
    \item $Pop_{scaled}$ --- нормализованная глобальная популярность трека.
    \item $\alpha$ --- балансировочный коэффициент.
\end{itemize}
\end{frame}

% === СЛАЙД 8: Метрики: HitRate@10 и NDCG@10 ===
\begin{frame}
\frametitle{Метрики: HitRate@10 и NDCG@10}
\textbf{Используемые метрики (Топ-10):}
\begin{itemize}
    \item \textbf{HitRate (HR):} Доля сессий, в которых скрытый целевой трек попал в итоговую выдачу. Формула основана на индикаторной функции: $HR = \frac{1}{N} \sum I(rank \le 10)$.
    \item \textbf{NDCG:} Оценивает качество ранжирования. Учитывает позицию релевантного трека через логарифмическое дисконтирование: знаменатель $\log_2(i + 1)$ штрафует алгоритм за низкие места в топе.
\end{itemize}
\end{frame}
